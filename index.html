Configuración Backend (Google Sheets)

Para que este sistema HTML funcione como un SaaS real y guarde todo en tu nube de Google, sigue estos pasos:

  1. Crea un archivo nuevo, totalmente en blanco, en Google Sheets.
  2. En el Sheets, ve arriba a Extensiones > Apps Script.
  3. Borra todo lo que salga ahí y pega el código de abajo.
  4. LA MAGIA: En la barra de arriba de Apps Script, asegúrate de que esté seleccionada la función setupAutomatizado y dale al botón Ejecutar ▶️. (Acepta los permisos de Google). ¡Ve a tu Excel y verás que el script armó todas las columnas por ti!
  5. Haz clic arriba a la derecha en Implementar > Nueva implementación. Elige Aplicación Web, y en acceso pon Cualquier persona.
  6. Copia la URL que te dará Google.
  7. Sube al recuadro blanco de arriba, pega esa URL y dale al botón "Conectar".
// 1. FUNCIÓN DE AUTO-CONSTRUCCIÓN (EJECUTAR 1 VEZ)
function setupAutomatizado() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  function prepararHoja(nombre, columnas) {
    var hoja = ss.getSheetByName(nombre) || ss.insertSheet(nombre);
    hoja.clear(); 
    hoja.getRange(1, 1, 1, columnas.length).setValues([columnas])
        .setFontWeight("bold").setBackground("#0f172a").setFontColor("#ffffff");
    hoja.setFrozenRows(1);
    for(var i=1; i<=columnas.length; i++) hoja.autoResizeColumn(i);
  }

  prepararHoja("Catalogo", ["ID", "Nombre", "Categoría", "Atributos (JSON)"]);
  prepararHoja("Compras", ["ID", "Fecha", "ID Producto", "Cantidad Inicial", "Cantidad Restante", "Costo Unitario ($)"]);
  prepararHoja("Ventas", ["ID", "Fecha", "ID Producto", "Cantidad Vendida", "Precio Venta ($)", "Costo FIFO ($)", "Ganancia Neta ($)"]);

  var h1 = ss.getSheetByName("Hoja 1"), h2 = ss.getSheetByName("Sheet1");
  if (h1 && ss.getSheets().length > 1) ss.deleteSheet(h1);
  if (h2 && ss.getSheets().length > 1) ss.deleteSheet(h2);
}

// 2. EL BACKEND (RECIBE DATOS DEL HTML)
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet();
    var data = JSON.parse(e.postData.contents);
    var tipo = data.tipo;
    
    if(tipo === "CATALOGO") {
      sheet.getSheetByName("Catalogo").appendRow([data.id, data.nombre, data.categoria, JSON.stringify(data.atributos)]);
    } else if(tipo === "COMPRA") {
      sheet.getSheetByName("Compras").appendRow([data.id, data.fecha, data.productoId, data.cantidad_inicial, data.cantidad_restante, data.costo]);
    } else if(tipo === "VENTA") {
      sheet.getSheetByName("Ventas").appendRow([data.id, data.fecha, data.productoId, data.cantidad, data.precio, data.costo_fifo, data.ganancia]);
    }
    return ContentService.createTextOutput(JSON.stringify({"status": "Exito"})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "Error", "msg": err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
