                return `

                        <td class="px-6 py-4 text-right font-medium text-rose-500">-$${v.costo_fifo.toFixed(2)}</td>

                        <td class="px-6 py-4 text-right font-black text-emerald-600">+$${v.ganancia.toFixed(2)}</td>

                `;

        function handleVentaSubmit(e) {

            e.preventDefault();

            const prodId = document.getElementById('vent-producto').value;

            const cantStr = document.getElementById('vent-cant').value;

            const precioStr = document.getElementById('vent-precio').value;

            const fecha = document.getElementById('vent-fecha').value;


            const qtyDeseada = parseInt(cantStr);

            const precio = parseFloat(precioStr);


            if(!prodId) return showToast('Seleccione un producto válido.', 'error');


            // Verificar Stock Total

            const inventarioDisponible = state.compras

                .filter(c => c.productoId === prodId)

                .reduce((acc, curr) => acc + curr.cantidad_restante, 0);


            if(inventarioDisponible < qtyDeseada) {

                return showToast(`Stock insuficiente. Solo tienes ${inventarioDisponible} unidades.`, 'error');

            }


            // ALGORITMO FIFO PUESTO EN PRÁCTICA

            let qtyRestante = qtyDeseada;

            let costoTotalFIFO = 0;

            

            // Ordenar por ID/Fecha (más viejo primero = FIFO)

            let lotesDelProducto = state.compras

                .filter(c => c.productoId === prodId && c.cantidad_restante > 0)

                .sort((a, b) => a.fecha.localeCompare(b.fecha) || a.id.localeCompare(b.id));


            for(let lote of lotesDelProducto) {

                if (qtyRestante <= 0) break;


                if (lote.cantidad_restante >= qtyRestante) {

                    // El lote cubre lo que falta

                    costoTotalFIFO += qtyRestante * lote.costo;

                    lote.cantidad_restante -= qtyRestante;

                    qtyRestante = 0;

                } else {

                    // El lote se agota, seguimos con el siguiente

                    costoTotalFIFO += lote.cantidad_restante * lote.costo;

                    qtyRestante -= lote.cantidad_restante;

                    lote.cantidad_restante = 0;

                }

            }


            // Registrar Venta

            const ingreso = qtyDeseada * precio;

            const ganancia = ingreso - costoTotalFIFO;


            const nuevaVenta = {

                id: `V-${Date.now()}`,

                fecha: fecha,

                productoId: prodId,

                cantidad: qtyDeseada,

                precio: precio,

                costo_fifo: costoTotalFIFO,

                ganancia: ganancia

            };


            state.ventas.push(nuevaVenta);

            simularEnvioSheet('VENTA', nuevaVenta);

            

            showToast(`Venta FIFO procesada. Ganancia: $${ganancia.toFixed(2)}`, 'success');

            document.getElementById('vent-cant').value = '';

            document.getElementById('vent-precio').value = '';

            

            renderVentas();

            updateMonthFilterOptions(); // Por si hay un nuevo mes

        }


        // --- LÓGICA: DASHBOARD Y ESTADÍSTICAS AVANZADAS ---

        function updateMonthFilterOptions() {

            const select = document.getElementById('global-month-filter');

            const currentValue = select.value;

            

            // Extraer meses únicos de ventas y compras (YYYY-MM)

            const meses = new Set();

            state.ventas.forEach(v => meses.add(v.fecha.substring(0, 7)));

            state.compras.forEach(c => meses.add(c.fecha.substring(0, 7)));

            

            // Añadir mes actual siempre

            const currentMonth = new Date().toISOString().substring(0, 7);

            meses.add(currentMonth);


            const mesesArray = Array.from(meses).sort().reverse(); // Más recientes primero

            

            let options = `<option value="ALL">Todo el tiempo</option>`;

            mesesArray.forEach(m => {

                const [year, month] = m.split('-');

                const nombreMes = new Date(year, parseInt(month)-1, 1).toLocaleString('es-ES', { month: 'long', year: 'numeric' });

                const label = m === currentMonth ? `Este Mes (${nombreMes})` : nombreMes;

                options += `<option value="${m}">${label.charAt(0).toUpperCase() + label.slice(1)}</option>`;

            });


            select.innerHTML = options;

            

            // Si es la primera carga, seleccionar el mes actual

            if(!currentValue || currentValue === '') {

                select.value = currentMonth;

            } else {

                select.value = currentValue; // Mantener selección previa si existe en la lista

            }

        }


        function renderDashboard() {

            const selectedMonth = document.getElementById('global-month-filter').value;

            

            // Filtrar datos

            const ventasFiltradas = selectedMonth === 'ALL' 

                ? state.ventas 

                : state.ventas.filter(v => v.fecha.startsWith(selectedMonth));

            

            // 1. KPIs

            const totalIngresos = ventasFiltradas.reduce((sum, v) => sum + (v.cantidad * v.precio), 0);

            const totalCostos = ventasFiltradas.reduce((sum, v) => sum + v.costo_fifo, 0);

            const beneficio = totalIngresos - totalCostos;

            const margen = totalIngresos > 0 ? (beneficio / totalIngresos) * 100 : 0;


            document.getElementById('dash-ingresos').innerText = `$${totalIngresos.toFixed(2)}`;

            document.getElementById('dash-costos').innerText = `$${totalCostos.toFixed(2)}`;

            document.getElementById('dash-beneficio').innerText = `$${beneficio.toFixed(2)}`;

            document.getElementById('dash-margen').innerText = `${margen.toFixed(1)}%`;


            // 2. Calcular datos para gráficos

            const categoryData = {};

            const productData = {}; // Para gráfico de barras (Ingreso vs Costo por prod)


            ventasFiltradas.forEach(v => {

                const prod = state.catalogo.find(p => p.id === v.productoId);

                if(!prod) return;

                

                const cat = prod.categoria;

                const pNombre = prod.nombre;

                const ingreso = v.cantidad * v.precio;


                // Categ

                if(!categoryData[cat]) categoryData[cat] = 0;

                categoryData[cat] += ingreso;


                // Prod

                if(!productData[pNombre]) productData[pNombre] = { ingreso: 0, costo: 0 };

                productData[pNombre].ingreso += ingreso;

                productData[pNombre].costo += v.costo_fifo;

            });


            // --- Renderizar Gráfico de Categorías (Doughnut) ---

            const catCtx = document.getElementById('chart-categories').getContext('2d');

            if(charts.categories) charts.categories.destroy();

            

            if(Object.keys(categoryData).length === 0) {

                 // Mock empty state

                 charts.categories = new Chart(catCtx, { type: 'doughnut', data: { labels: ['Sin datos'], datasets: [{ data: [1], backgroundColor: ['#f1f5f9'] }] }, options: { cutout: '70%', plugins: { legend: { display: false }, tooltip: { enabled: false } } }});

            } else {

                charts.categories = new Chart(catCtx, {

                    type: 'doughnut',

                    data: {

                        labels: Object.keys(categoryData),

                        datasets: [{

                            data: Object.values(categoryData),

                            backgroundColor: ['#14b8a6', '#0ea5e9', '#f59e0b', '#8b5cf6', '#ec4899'],

                            borderWidth: 0,

                            hoverOffset: 4

                        }]

                    },

                    options: {

                        responsive: true, maintainAspectRatio: false, cutout: '70%',

                        plugins: { legend: { position: 'bottom', labels: { usePointStyle: true, boxWidth: 8, font: {family: 'Inter', size: 11} } } }

                    }

                });

            }


            // --- Renderizar Gráfico de Productos (Bar) ---

            const prodCtx = document.getElementById('chart-products').getContext('2d');

            if(charts.products) charts.products.destroy();


            const pLabels = Object.keys(productData);

            const pIngresos = pLabels.map(l => productData[l].ingreso);

            const pCostos = pLabels.map(l => productData[l].costo);


            charts.products = new Chart(prodCtx, {

                type: 'bar',

                data: {

                    labels: pLabels.length > 0 ? pLabels : ['Sin datos'],

                    datasets: [

                        { label: 'Ingreso Bruto', data: pLabels.length > 0 ? pIngresos : [0], backgroundColor: '#14b8a6', borderRadius: 4 },

                        { label: 'Costo FIFO', data: pLabels.length > 0 ? pCostos : [0], backgroundColor: '#cbd5e1', borderRadius: 4 }

                    ]

                },

                options: {

                    responsive: true, maintainAspectRatio: false,

                    scales: { 

                        y: { beginAtZero: true, grid: { borderDash: [4, 4], color: '#f1f5f9' }, ticks: { font: {family: 'Inter', size: 10} } },

                        x: { grid: { display: false }, ticks: { font: {family: 'Inter', size: 10} } }

                    },

                    plugins: { legend: { position: 'top', align: 'end', labels: { boxWidth: 12, font: {family: 'Inter', size: 11} } } }

                }

            });


            // 3. Tabla de Stock Consolidado (Global, no afectada por el filtro de mes)

            const stockConsolidado = state.catalogo.map(p => {

                const lotes = state.compras.filter(c => c.productoId === p.id);

                const stock = lotes.reduce((acc, c) => acc + c.cantidad_restante, 0);

                const valor = lotes.reduce((acc, c) => acc + (c.cantidad_restante * c.costo), 0);

                return { ...p, stock, valor };

            }).filter(p => p.stock > 0).sort((a, b) => b.valor - a.valor); // Ordenar por valor mayor


            const stockBody = document.getElementById('dash-stock-body');

            if(stockConsolidado.length === 0) {

                stockBody.innerHTML = `<tr><td colspan="4" class="px-5 py-8 text-center text-slate-400 italic">No hay inventario valorizado actualmente.</td></tr>`;

            } else {

                stockBody.innerHTML = stockConsolidado.map(s => `

                    <tr class="hover:bg-slate-50">

                        <td class="px-5 py-3 font-bold text-slate-800">${s.nombre}</td>

                        <td class="px-5 py-3"><span class="bg-slate-100 text-slate-600 px-2 py-1 rounded text-[10px] font-bold uppercase tracking-wider">${s.categoria}</span></td>

                        <td class="px-5 py-3 text-center">

                            <span class="bg-brand-50 text-brand-700 font-bold px-3 py-1 rounded-full text-xs">${s.stock} u.</span>

                        </td>

                        <td class="px-5 py-3 text-right font-black text-slate-700">$${s.valor.toFixed(2)}</td>

                    </tr>

                `).join('');

            }

        }
